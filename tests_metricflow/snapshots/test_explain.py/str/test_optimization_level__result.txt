O4:
  SELECT
    COALESCE(nr_subq_6.metric_time__day, nr_subq_13.metric_time__day) AS metric_time__day
    , COALESCE(nr_subq_6.listing__country_latest, nr_subq_13.listing__country_latest) AS listing__country_latest
    , MAX(nr_subq_6.bookings) AS bookings
    , MAX(nr_subq_13.views) AS views
  FROM (
    SELECT
      nr_subq_0.metric_time__day AS metric_time__day
      , listings_latest_src_10000.country AS listing__country_latest
      , SUM(nr_subq_0.bookings) AS bookings
    FROM (
      SELECT
        DATE_TRUNC('day', ds) AS metric_time__day
        , listing_id AS listing
        , 1 AS bookings
      FROM ***************************.fct_bookings bookings_source_src_10000
    ) nr_subq_0
    LEFT OUTER JOIN
      ***************************.dim_listings_latest listings_latest_src_10000
    ON
      nr_subq_0.listing = listings_latest_src_10000.listing_id
    GROUP BY
      nr_subq_0.metric_time__day
      , listings_latest_src_10000.country
  ) nr_subq_6
  FULL OUTER JOIN (
    SELECT
      nr_subq_7.metric_time__day AS metric_time__day
      , listings_latest_src_10000.country AS listing__country_latest
      , SUM(nr_subq_7.views) AS views
    FROM (
      SELECT
        DATE_TRUNC('day', ds) AS metric_time__day
        , listing_id AS listing
        , 1 AS views
      FROM ***************************.fct_views views_source_src_10000
    ) nr_subq_7
    LEFT OUTER JOIN
      ***************************.dim_listings_latest listings_latest_src_10000
    ON
      nr_subq_7.listing = listings_latest_src_10000.listing_id
    GROUP BY
      nr_subq_7.metric_time__day
      , listings_latest_src_10000.country
  ) nr_subq_13
  ON
    (
      nr_subq_6.listing__country_latest = nr_subq_13.listing__country_latest
    ) AND (
      nr_subq_6.metric_time__day = nr_subq_13.metric_time__day
    )
  GROUP BY
    COALESCE(nr_subq_6.metric_time__day, nr_subq_13.metric_time__day)
    , COALESCE(nr_subq_6.listing__country_latest, nr_subq_13.listing__country_latest)
