sql_with_cte:
  -- Combine Aggregated Outputs
  -- Write to DataTable
  WITH cm_0_cte AS (
    -- Aggregate Inputs for Simple Metrics
    -- Compute Metrics via Expressions
    SELECT
      metric_time__day
      , SUM(__bookings) AS bookings
    FROM (
      -- Read Elements From Semantic Model 'bookings_source'
      -- Metric Time Dimension 'ds'
      -- Select: ['__bookings', 'metric_time__day']
      -- Select: ['__bookings', 'metric_time__day']
      SELECT
        DATE_TRUNC('day', ds) AS metric_time__day
        , 1 AS __bookings
      FROM ***************************.bookings_source bookings_source_src_34000
    ) subq_19
    GROUP BY
      metric_time__day
  )

  , cm_1_cte AS (
    -- Compute Metrics via Expressions
    SELECT
      metric_time__day
      , bookings AS metric_level_0_index_0
    FROM (
      -- Read From CTE For node_id=cm_0
      SELECT
        metric_time__day
        , bookings
      FROM cm_0_cte
    ) subq_21
  )

  , cm_2_cte AS (
    -- Compute Metrics via Expressions
    SELECT
      metric_time__day
      , bookings AS metric_level_0_index_1
    FROM (
      -- Read From CTE For node_id=cm_0
      SELECT
        metric_time__day
        , bookings
      FROM cm_0_cte
    ) subq_23
  )

  SELECT
    COALESCE(subq_26.metric_time__day, subq_30.metric_time__day) AS metric_time__day
    , MAX(subq_26.metric_level_1_index_0) AS metric_level_1_index_0
    , MAX(subq_30.metric_level_1_index_1) AS metric_level_1_index_1
  FROM (
    -- Compute Metrics via Expressions
    SELECT
      metric_time__day
      , metric_level_0_index_0 + metric_level_0_index_1 AS metric_level_1_index_0
    FROM (
      -- Combine Aggregated Outputs
      SELECT
        COALESCE(cm_1_cte.metric_time__day, cm_2_cte.metric_time__day) AS metric_time__day
        , MAX(cm_1_cte.metric_level_0_index_0) AS metric_level_0_index_0
        , MAX(cm_2_cte.metric_level_0_index_1) AS metric_level_0_index_1
      FROM cm_1_cte
      FULL OUTER JOIN
        cm_2_cte
      ON
        cm_1_cte.metric_time__day = cm_2_cte.metric_time__day
      GROUP BY
        COALESCE(cm_1_cte.metric_time__day, cm_2_cte.metric_time__day)
    ) subq_25
  ) subq_26
  FULL OUTER JOIN (
    -- Compute Metrics via Expressions
    SELECT
      metric_time__day
      , metric_level_0_index_0 + metric_level_0_index_1 AS metric_level_1_index_1
    FROM (
      -- Combine Aggregated Outputs
      SELECT
        COALESCE(cm_1_cte.metric_time__day, cm_2_cte.metric_time__day) AS metric_time__day
        , MAX(cm_1_cte.metric_level_0_index_0) AS metric_level_0_index_0
        , MAX(cm_2_cte.metric_level_0_index_1) AS metric_level_0_index_1
      FROM cm_1_cte
      FULL OUTER JOIN
        cm_2_cte
      ON
        cm_1_cte.metric_time__day = cm_2_cte.metric_time__day
      GROUP BY
        COALESCE(cm_1_cte.metric_time__day, cm_2_cte.metric_time__day)
    ) subq_29
  ) subq_30
  ON
    subq_26.metric_time__day = subq_30.metric_time__day
  GROUP BY
    COALESCE(subq_26.metric_time__day, subq_30.metric_time__day)
