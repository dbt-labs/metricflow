sql_without_cte:
  -- Compute Metrics via Expressions
  -- Write to DataTable
  SELECT
    metric_time__day
    , listing__country_latest
    , bookings
    , views
    , CAST(bookings AS DOUBLE) / CAST(NULLIF(views, 0) AS DOUBLE) AS bookings_per_view
  FROM (
    -- Combine Aggregated Outputs
    SELECT
      COALESCE(subq_9.metric_time__day, subq_18.metric_time__day) AS metric_time__day
      , COALESCE(subq_9.listing__country_latest, subq_18.listing__country_latest) AS listing__country_latest
      , MAX(subq_9.bookings) AS bookings
      , MAX(subq_18.views) AS views
    FROM (
      -- Join Standard Outputs
      -- Select: ['__bookings', 'listing__country_latest', 'metric_time__day']
      -- Select: ['__bookings', 'listing__country_latest', 'metric_time__day']
      -- Aggregate Inputs for Simple Metrics
      -- Compute Metrics via Expressions
      SELECT
        subq_1.metric_time__day AS metric_time__day
        , listings_latest_src_28000.country AS listing__country_latest
        , SUM(subq_1.__bookings) AS bookings
      FROM (
        -- Read Elements From Semantic Model 'bookings_source'
        -- Metric Time Dimension 'ds'
        SELECT
          DATE_TRUNC('day', ds) AS metric_time__day
          , listing_id AS listing
          , 1 AS __bookings
        FROM ***************************.fct_bookings bookings_source_src_28000
      ) subq_1
      LEFT OUTER JOIN
        ***************************.dim_listings_latest listings_latest_src_28000
      ON
        subq_1.listing = listings_latest_src_28000.listing_id
      GROUP BY
        subq_1.metric_time__day
        , listings_latest_src_28000.country
    ) subq_9
    FULL OUTER JOIN (
      -- Join Standard Outputs
      -- Select: ['__views', 'listing__country_latest', 'metric_time__day']
      -- Select: ['__views', 'listing__country_latest', 'metric_time__day']
      -- Aggregate Inputs for Simple Metrics
      -- Compute Metrics via Expressions
      SELECT
        subq_11.metric_time__day AS metric_time__day
        , listings_latest_src_28000.country AS listing__country_latest
        , SUM(subq_11.__views) AS views
      FROM (
        -- Read Elements From Semantic Model 'views_source'
        -- Metric Time Dimension 'ds'
        SELECT
          DATE_TRUNC('day', ds) AS metric_time__day
          , listing_id AS listing
          , 1 AS __views
        FROM ***************************.fct_views views_source_src_28000
      ) subq_11
      LEFT OUTER JOIN
        ***************************.dim_listings_latest listings_latest_src_28000
      ON
        subq_11.listing = listings_latest_src_28000.listing_id
      GROUP BY
        subq_11.metric_time__day
        , listings_latest_src_28000.country
    ) subq_18
    ON
      (
        subq_9.listing__country_latest = subq_18.listing__country_latest
      ) AND (
        subq_9.metric_time__day = subq_18.metric_time__day
      )
    GROUP BY
      COALESCE(subq_9.metric_time__day, subq_18.metric_time__day)
      , COALESCE(subq_9.listing__country_latest, subq_18.listing__country_latest)
  ) subq_19

      listing_id AS listing

    metric_time__day AS metric_time__day
    , listing__country_latest AS listing__country_latest
